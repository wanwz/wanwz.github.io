非常简单的一个脚本用来修改主机名。
<!-- more -->

```
#!/bin/bash
#Author: wanwz
#Date: 2019-12-25
#Comment: 修改主机名
version=`cat /etc/redhat-release | sed -r 's/.* ([0-9]+)\..*/\1/'`
case ${version} in
	[7])
		echo "***当前系统版本是7***"
		read -p "修改主机名为：" cname
		hostnamectl set-hostname $cname
		sed -i 's/localhost4.localdomain4/localhost4.localdomain4 $cname/' /etc/hosts
		if [ $?==0 ];then
			echo "修改成功"
		else
			echo "修改失败"
		fi
;;	
	[6])
		echo "***当前系统版本是6***"
		read -p "修改主机名为：" cname
		sed -i 's/HOSTNAME=/HOSTNAME=$cname/' /etc/sysconfig/network
		sed -i 's/localhost4.localdomain4/localhost4.localdomain4 $cname/' /etc/hosts
		if [ $?==0 ];then
			echo "修改成功"
		else
			echo "修改失败"
		fi
;;
	*)
		echo "其它系统版本！"
esac
```
